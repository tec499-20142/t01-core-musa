\usecase{Buscar Instrução}

O módulo \textbf{Instruction Fetch} é responsável por buscar a instrução a ser executada.

 
\actors
\begin{description}
 \actor{Instruction Fetch}{Módulo responsável por buscar a instrução a ser executada.}
 \actor{Unit Control}{Unidade que controla a execução das operações.}
\end{description}

\preconditions 
\begin{itemize}
 \item O módulo \textbf{Instruction Fetch} precisa estar habilitado pela \textbf{Unit Control}.
 \item Endereço da instrução a ser buscada.
\end{itemize}

\postconditions
\begin{itemize}
  \item Enviar a instrução encontrada para o \textbf{Instruction Decode}.
\end{itemize}
